#正则表达式
.   #匹配除换行符以外的任意字符
\w  #匹配数字、字母或下划线
\s  #匹配任意的空白符
\d  #匹配数字
\b  #匹配单词的开头或结尾
^   #匹配字符串的开始
$   #匹配字符串的结束
\n  #换行符
\t  #制表符
[aeiou]     #匹配aeiou中的任意一个字符
[^aeiou]    #匹配除了aeiou以外的任意字符
[a-f]       #匹配a到f中的任意一个字符
[a-fl]      #匹配a到f和l中的任意一个字符，即abcdefl中的任意一个字符
[a-f]+      #匹配a到f中的任意多个字符，此时正则匹配是贪婪的，即匹配尽可能多的字符
[a-f]*      #匹配a到f中的任意多个字符，包括0个，此时正则匹配是贪婪的，即匹配尽可能多的字符
[a-f]?      #匹配a到f中的任意一个字符，包括0个
[a-f]+?     #匹配a到f中的任意多个字符，此时正则匹配是非贪婪的，即匹配尽可能少的字符
[a-f]{3}    #匹配a到f中的3个字符
(a|bc)      #匹配a或者bc
